/* Export Control Sheet to Krust */ 



%ImportControlSheet;

data krust.ControlSheet;
set KrustControlSheet;
run;

/*if not (year=2013 and month=3);*/

title "Sales Volumes in Control Sheet ";
proc sql;
select prodcat, year, month, sum(net) as Sales
from krust.ControlSheet
group by prodcat, year , month 
order by prodcat, year desc, month desc
;
quit;
