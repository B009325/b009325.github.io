/*---------------------- Raider Macro ----------------------*/
* If ExcludeRaiders is true then it will strip them from the file;
%macro RemoveR(xFiles);

proc sort data=&xFiles.;
by Custid;
run;

data &xFiles.; 
merge &xFiles. (in=a) itch.RR_Customers (in=b);
by custid;
if a and not b;
run;

%mend;
